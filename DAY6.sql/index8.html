Câu lệnh DELETE ( xoá dữ liệu )
        cú pháp : DELETE FROM table_name WHERE condition;
        ví dụ : DELETE FROM customers WHERE id = 30
        ( xoá bản ghi có id = 1 )


        .
        1 - từ khoá MIN lấy giá trị nhỏ nhất
        ví dụ :
        SELECT MIN(id) FROM customers
        lấy ra id nhỏ nhất trong bảng

        2 - từ khoá MAX lấy giá trị lớn nhất
        SELECT MAX(id) FROM customers
        lấy ra id lớn nhất trong bảng
