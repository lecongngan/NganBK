WHERE CustomerName LIKE 'a%'	tìm tất cả các giá trị bắt đầu là "a"
        WHERE CustomerName LIKE '%a'	tìm tất cả các giá trị kết thúc là a
        WHERE CustomerName LIKE '%or%'	tìm tất cả các giá trị chứa "or"
        WHERE CustomerName LIKE '_r%'	tim tất cả các giá trị mà "r" là chữ thứ 2
        WHERE CustomerName LIKE 'a_%_%'	tìm tất cả các giá trị bắt đầu là a và có ít nhất 3 ký tự
        WHERE ContactName LIKE 'a%o'	giá trị bắt đầu là :"a" kết thúc là "o"

        Ví dụ
        SELECT * FROM customers WHERE first_name LIKE 'j%'
        tìm tất cả các khách hàng bắt đầu là chữ j
